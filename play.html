function listenToSpecificGame(gameId) { // Verwijder een eventuele oude actieve inschrijving om dubbele luisteraars te voorkomen if (gameSubscription) { supabaseClient.removeChannel(gameSubscription); } // Maak een kanaal aan dat specifiek luistert naar updates in de gcvm_games tabel gameSubscription = supabaseClient.channel(`game_updates_${gameId}`) .on( 'postgres_changes', { event: 'UPDATE', schema: 'public', table: 'gcvm_games', filter: `id=eq.${gameId}` }, (payload) => { const updatedGame = payload.new; if (updatedGame.status === 'playing') { // Als de speler nog in de lobby staat, stuur hem direct door naar de live match if (document.getElementById('game-view').classList.contains('hidden')) { initGame(updatedGame); return; } // HARD SYNC: Overschrijf altijd de lokale engine met de FEN uit de database game.load(updatedGame.fen); board.position(updatedGame.fen); // Synchroniseer de resterende kloktijden direct met de database whiteTimeLeft = updatedGame.white_time; blackTimeLeft = updatedGame.black_time; // Corrigeer eventuele netwerkvertraging (verstreken seconden sinds de zet) adjustTimeForPassedLapse(updatedGame.last_move_at, updatedGame.fen); // Controleer op remise-aanbiedingen van de tegenstander if (updatedGame.draw_offer && updatedGame.draw_offer !== myColor) { document.getElementById('accept-draw-btn').classList.remove('hidden'); } else if (!updatedGame.draw_offer) { document.getElementById('accept-draw-btn').classList.add('hidden'); } // Werk de UI-elementen direct bij updateClockDisplays(); updateTurnIndicator(); // Start of herstart de lokale aftelklok voor de speler die nu aan de beurt is startLocalClock(); } else if (updatedGame.status.endsWith('_won') || updatedGame.status === 'draw') { clearInterval(clockInterval); game.load(updatedGame.fen); board.position(updatedGame.fen); displayGameEndMessage(updatedGame.status); } } ) .subscribe((status) => { if (status === 'SUBSCRIBED') { console.log(`Succesvol verbonden met live game-room: ${gameId}`); } }); }
